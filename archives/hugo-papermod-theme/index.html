根目录新建layouts\partials\comments.html：

<script src="https://cdn.jsdelivr.net/npm/@waline/client"></script>
<div id="waline"></div>
<script>
  new Waline({
    el: '#waline',
    serverURL: 'https://your-domain.vercel.app',
    copyright: true,
    login: 'enable',
    placeholder: '欢迎评论',
    dark: '.dark',
	requiredMeta: ['nick', 'mail'],
  });
</script>
